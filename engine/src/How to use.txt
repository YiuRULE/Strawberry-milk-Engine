#include "ThreadPool.h"
#include <iostream>

void show()
{
	int i = 1;
	while (i < 12)
	{
	std::cout << "Suis pas carre" << std::endl;
	i++;
}
}

void showtoo()
{
	int i = 1;
	while (i < 12)
	{
		std::cout << "Suis pas rond" << std::endl;
		i++;
	}
}


int main()
{
